# AC

system-view
sysname Company_1-AC     

dhcp enable                
vlan 100
	quit

vlan pool Company_1
	vlan 10 20 30 40 50 60 70
	quit

interface Vlanif 100 
 ip address 10.1.100.254 24 
 dhcp select interface

quit

 port link-type trunk
 port trunk allow-pass vlan 100
quit

# 使用管理口作为源接口
